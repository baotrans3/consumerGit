<script type="text/javascript">
    AutoSyncLicenseStatus = {
        timeCallServer: 30000,
        maximumCallServer: 3,
        countCallServer: 0,
        isChanged: false,
        getLicenseStolen: function () {
            var status = 'Stolen';
            var license = '';
            $('td.device-status').each(function () {
                if ($.trim($(this).html()) === status) {
                    var id = $(this).closest('tr')[0].getAttribute('onclick');
                    if (id.length > 21) {
                        id = id.replace('javascript:rowClick(', '').replace(');', '');
                        license += id + ',';
                    }
                }
            });
            if (license.length > 0) {
                license = license.substr(0, license.length - 1);
            }
            return license;
        },
        autoSync: function () {
            var self = this,
            license = self.getLicenseStolen();
            if (license && license.length > 0 && self.countCallServer < self.maximumCallServer) {
                $.ajax({
                    url: '/ChooseDevice/CheckStolenChanged',
                    type: 'POST',
                    data: JSON.stringify({ stolenLicenses: license, fromView: 'ChooseDevice' }),
                    contentType: "application/json",
                    dataType: "json",
                    success: function (data, status) {
                        if (data && data.isValid && data.isChanged) {
                            var searchCondition = 0;
                            var keySearch = '';
                            if (typeof $('#callSearchDevices').attr('keySearch') != "undefined" && $('#callSearchDevices').attr('keySearch').length > 0) {
                                keySearch = $('#callSearchDevices').attr('keySearch');
                            }
                            searchCondition = (typeof $('#callSearchDevices').attr('searchOption') === "undefined") ? 0 : $('#callSearchDevices').attr('searchOption');
                            if (AutoSyncLicenseStatus.currentPage == 1 && keySearch == '' && searchCondition == 0) {
                                location.href = window.location.href;
                            }
                            else {
                                LoadPaging(AutoSyncLicenseStatus.currentPage, 'paging', 'ChooseDevice');
                            }
                        } else {
                            self.countCallServer++;
                            setTimeout("AutoSyncLicenseStatus.autoSync()", self.timeCallServer);
                        }
                    },
                    error: function (xhr, ajaxOptions, thrownError) {
                        self.countCallServer++;
                        setTimeout("AutoSyncLicenseStatus.autoSync()", self.timeCallServer);
                    }
                });

            }

        },
        currentPage: 1 // first page.
    }
    
    function checkPagingTimeOut() {
        $.ajax({
            url: './Home/CheckLogin',
            type: 'POST',
            success: function (data, textStatus, xhr) {
                if (data.state == "timeout") {
                    window.location.href = data.rtnValue;
                }
                if (data.state == "logout") {
                    window.location.href = data.rtnValue;
                }
            },
            error: function (xhr, textStatus, thrownError) {
                window.location.href = "/Home/Index?messageId=sessionTimeout";
            }
        });
    }

    function LoadPaging(index, id, controller) {
        checkPagingTimeOut();
        window.setTimeout(function () {
            var searchCondition = 0;
            var keySearch = '';
            if (typeof $('#callSearchDevices').attr('keySearch') != "undefined" && $('#callSearchDevices').attr('keySearch').length > 0) {
                keySearch = $('#callSearchDevices').attr('keySearch');
            }
            searchCondition = (typeof $('#callSearchDevices').attr('searchOption') === "undefined") ? 0 : $('#callSearchDevices').attr('searchOption');
            // keep page index when page is refresh.
            AutoSyncLicenseStatus.currentPage = index;
            var dataJson = { pageIndex: index, searchCondition: searchCondition, keySearch: keySearch };
            var jsonSerialised = JSON.stringify(dataJson);
            $.ajax({
                url: '/ChooseDevice',
                type: 'post',
                data: jsonSerialised,
                contentType: 'application/json; charset=utf-8',
                success: function (data, status) {
                    if (status == "success") {
                        var dom = $(data);
                        $('#' + id).empty().append(dom.find('#' + id).html());
                        $('#keySearch').val($('#callSearchDevices').attr('keySearch'));
                        $('#searchOption option').each(function () {
                            if ($(this).val() == searchCondition) {
                                this.selected = true;
                            }
                        });
                        AddRounds();
                    }
                }
            });
        }, 1000);
    }

    function showError() {
        if ($('#paging tr').length === 1) {
            $('#paging tr th').remove();
            $('#paging tr').append('<td>No match</td>');
        }
    }
    function callSearch() {
        $('#callSearchDevices').click();
    }

    $(function () {
        $('#callSearchDevices').click(function () {
            checkPagingTimeOut();
            window.setTimeout(function () {
                $('#callSearchDevices').attr('keySearch', $.trim($('#keySearch').val())); // store data for key search
                $('#callSearchDevices').attr('searchOption', $('#searchOption').val()); // store data for search option
                AutoSyncLicenseStatus.currentPage = 1;                                   // reset current page is 1
                var dataJson = { pageIndex: 1, searchCondition: $('#searchOption').val(), keySearch: $.trim($('#keySearch').val()) };
                var jsonSerialised = JSON.stringify(dataJson);
                $.ajax({
                    url: '/ChooseDevice',
                    type: 'post',
                    data: jsonSerialised,
                    contentType: 'application/json; charset=utf-8',
                    success: function (data, status) {
                        if (status == "success") {
                            var dom = $(data);
                            $('#paging').empty().append(dom.find('#paging').html());
                            showError();
                            $('#keySearch').val($('#callSearchDevices').attr('keySearch'));
                            $('#searchOption option').each(function () {
                                if ($(this).val() == $('#callSearchDevices').attr('searchOption')) {
                                    this.selected = true;
                                }
                            });
                            AddRounds();
                        }
                    }
                });
            }, 1000);
        });

        $('#rowHiddenClick').click(function () {
            window.location.href = '/ChooseDevice/Device?licenseId=' + $(this).attr('licenseId');
            return false;
        });

        AddRounds();

        // limited character of device's names are too long.
        AdjustLengthOfDeviceName();
    });

    function AddRounds() {
        //Auto Sync license status.
        AutoSyncLicenseStatus.countCallServer = 0;
        AutoSyncLicenseStatus.autoSync();
        
        if ($.browser.msie) {

                    $('#btnSearch').addClass('save-button');
                    $('#btnSearch').addClass('solo-button-ie');
                    $('#btnSearch').addClass('submitFormClass');
                    $('#btnSearch').attr('id', 'btnSearch-ie');
                }
            }
        }
    }

    function rowClick(licensId) {

    function AdjustLengthOfDeviceName(){
         var culture = "en-us";
         if (culture.length > 0 && (culture.toLowerCase() == "de-de" || culture.toLowerCase() == "sv-se")) {
             var labelDeviceNames = $('table.switch').find('tr.cursor td.device-name label');
             if (labelDeviceNames.length > 0) {
                 $(labelDeviceNames).each(function () {
                     var deviceName = $(this);
                     var deviceNameArr = $.trim($(this).text()).split(' ');
                     if (deviceNameArr.length > 0) {
                         $(deviceNameArr).each(function (index) {
                             if (deviceNameArr[index].length >= 20) {
                                 var shortName = deviceName.text().substr(0, 18) + "...";
                                 $(deviceName).text(shortName);
                             }
                         });
                     }
                 });
             }
         }
    }

    function SetCookie(name, value) {
        document.cookie = name + "=" + value + "; path=/";
        if (window.parent) {
            window.parent.document.cookie = name + "=" + value + "; path=/"; 
        } 
    }

    function GetCookie(name) {
        var nameEQ = name + "=";
        var ca = document.cookie.split(';');
        for (var i = 0; i < ca.length; i++) {
            var c = ca[i];
            while (c.charAt(0) == ' ') c = c.substring(1, c.length);
            if (c.indexOf(nameEQ) == 0)
                return c.substring(nameEQ.length, c.length);
        }
        return null;
    }

    function ClearCookie(name) {
        document.cookie = name + "=''; path=/";
    }
